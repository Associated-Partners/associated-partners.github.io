\begin{frame}{Goal}
  \begin{block}{Immediate objective}
    Make \texttt{assocpartners.com} look legitimate, polished, and
    business-ready --- while keeping the website maintainable from one
    GitHub repository and leaving room for a private assistant later.
  \end{block}

  \vspace{0.6em}
  \textbf{Do not build yet:} CRM, events database, booking system,
  CMS, autonomous email sending, or domain-wide impersonation.
\end{frame}
